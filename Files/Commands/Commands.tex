
% command til at lave et billede
% #1: placering af billede
% #2: width/scale
% #3: sti til billede
% #4: billede text
% #5: label
\newcommand{\pic}[5]
{
\begin{figure} [#1]
\includegraphics[width=#2]{#3}
\caption{#4}
\label{#5}
\end{figure}
}
